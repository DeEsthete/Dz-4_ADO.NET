CREATE DATABASE SellingDB
USE SellingDB

CREATE TABLE Buyers(
Id INT NOT NULL IDENTITY PRIMARY KEY,
Name VARCHAR(50) NOT NULL,
Surname VARCHAR(50) NOT NULL
)

CREATE TABLE Sellers(
Id INT NOT NULL IDENTITY PRIMARY KEY,
Name VARCHAR(50) NOT NULL,
Surname VARCHAR(50) NOT NULL
)

CREATE TABLE Orders(
Id INT NOT NULL IDENTITY PRIMARY KEY,
Buyers_id INT FOREIGN KEY REFERENCES Buyers(Id) NOT NULL,
Sellers_id INT FOREIGN KEY REFERENCES Sellers(Id) NOT NULL,
Cost INT NOT NULL,
Order_date DATETIME NOT NULL,
)
